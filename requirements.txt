# =============================================================================
# sqmg_project-cudaq — 環境依賴
# 針對 DGX V100（Volta, sm_70）修正版
# =============================================================================
#
# !! 重要說明 !!
# CUDA-Q >= 0.8.0 的 pip wheel 預編譯 PTX 僅涵蓋 Ampere(sm_80)/Hopper(sm_90)，
# 不支援 V100 Volta(sm_70)，呼叫 cudaq.set_target("nvidia") 時會靜默 fallback
# 到 qpp-cpu（CPU 模擬），導致 GPU 完全未被使用。
#
# 解法：降版至 0.7.1，此版本仍包含 sm_70 的預編譯 SASS。
#
# 安裝指令（依 DGX CUDA Toolkit 版本選一）：
#   CUDA 11.x：pip install cuda-quantum-cu11==0.7.1
#   CUDA 12.x：pip install cuda-quantum-cu12==0.7.1
#
# 確認 CUDA 版本：nvcc --version 或 nvidia-smi
# =============================================================================

# ── CUDA-Q 核心引擎（V100 sm_70 相容版）────────────────────────────────────
# 注意：pip 套件名稱為 cuda-quantum-cu11 / cuda-quantum-cu12，import 仍為 cudaq
# 此行僅作版本紀錄，實際安裝請用下方「安裝指令」手動執行
cuda-quantum-cu12==0.7.1

# ── 化學/分子生成 ────────────────────────────────────────────────────────────
rdkit>=2023.9.5      # 0.7.1 時代對應版本，避免 API 不相容

# ── 數值計算與資料處理 ───────────────────────────────────────────────────────
numpy>=1.24,<2.0     # cudaq 0.7.x 與 numpy 2.x 有已知相容問題，限定 1.x
pandas>=2.0

# ── 視覺化 ───────────────────────────────────────────────────────────────────
matplotlib>=3.7
seaborn>=0.12

# ── 機器學習 ─────────────────────────────────────────────────────────────────
scikit-learn>=1.3
