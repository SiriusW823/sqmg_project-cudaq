\node[stageO] (o1) at (6.7,-2.9)

\node[stageO,below=6mm of o1] (o2)

\node[stageO,below=6mm of o2] (o3)
  {\textbf{Stagnation \& diversity control}\\ reinit 25\% worst after 12 stalls; mode-collapse recycling};
\node[stageO,below=6mm of o3] (o4)
  {\textbf{Chemistry constraint}\\ \texttt{ConditionalWeightsGenerator}\\ constrains the 134 weights before dispatch};

% ===================== Eval stages =====================
\node[stageQ] (e1) at (13.9,-2.9)

\node[stageQ,below=6mm of e2] (e3)
